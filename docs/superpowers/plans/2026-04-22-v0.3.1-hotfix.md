- [ ] **Step 3: Implement honest case resolution in the validator**

```python
def _resolve_subset_cases(
    *,
    subset_name: str,
    subset: dict[str, Any],
    skill_dir: Path,
    skill_id: str,
    errors: list[str],
) -> list[str]:
    raw_cases = subset.get("cases", [])
    if not isinstance(raw_cases, list):
        errors.append(f"{skill_id}: {subset_name} cases must be a list")
        return []

    resolved: list[str] = []
    for entry in raw_cases:
        if not isinstance(entry, str):
            errors.append(f"{skill_id}: {subset_name} contains non-string case entry {entry!r}")
            continue
        if any(ch in entry for ch in "*?[]"):
            matches = sorted((skill_dir / entry).parent.glob(Path(entry).name))
            if not matches:
                errors.append(
                    f"{skill_id}: {subset_name} cases pattern matched 0 files: {entry}"
                )
                continue
            resolved.extend(str(path.resolve()) for path in matches)
        else:
            resolved.append(entry)
    return resolved


def _subset_case_count(
    *,
    subset_name: str,
    subset: dict[str, Any],
    skill_dir: Path,
    skill_id: str,
    errors: list[str],
) -> int:
    resolved_cases = _resolve_subset_cases(
        subset_name=subset_name,
        subset=subset,
        skill_dir=skill_dir,
        skill_id=skill_id,
        errors=errors,
    )
    total = subset.get("total")
    if total is not None and int(total) != len(resolved_cases):
        errors.append(
            f"{skill_id}: {subset_name} total={total} does not match resolved_cases={len(resolved_cases)}"
        )
    return len(resolved_cases)
```
